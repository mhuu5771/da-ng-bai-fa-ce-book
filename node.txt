    <script>
        let store = {};

        function generateStandardLink(pageId) {
            return /^\d+$/.test(pageId) ? `https://www.facebook.com/profile.php?id=${pageId}` : `https://www.facebook.com/${pageId}`;
        }

        function addLog(msg, type, url = "#") {
            const body = document.getElementById('history-body');
            if (body.innerHTML.includes("Chưa có dữ liệu")) body.innerHTML = "";
            const tr = document.createElement('tr');
            tr.className = "hover:bg-white/5 transition-all group";
            const statusStyle = type === 'success' ? 'text-sky-400 bg-sky-400/10 border-sky-400/20' : (type === 'schedule' ? 'text-amber-400 bg-amber-400/10 border-amber-400/20' : 'text-red-400 bg-red-400/10 border-red-400/20');
            tr.innerHTML = `
                <td class="px-6 py-3 text-slate-500">${new Date().toLocaleTimeString()}</td>
                <td class="px-6 py-3 text-center">
                    <span class="px-2 py-1 rounded-full border ${statusStyle} text-[8px] font-black uppercase tracking-widest inline-block">${msg}</span>
                </td>
                <td class="px-6 py-3 text-right">
                    ${url !== "#" ? `<a href="${url}" target="_blank" class="inline-flex items-center gap-2 bg-sky-600 text-white px-3 py-1 rounded-full text-[8px] font-black uppercase hover:bg-sky-500 transition-all shadow-lg">Xem chi tiết</a>` : '<span class="text-slate-800">---</span>'}
                </td>
            `;
            body.prepend(tr);
        }

        function syncAllLinks() {
            const sel = document.getElementById('p-select');
            if(!sel.value) return;
            const standardUrl = generateStandardLink(sel.value);
            const count = parseInt(document.getElementById('card-count').value);
            for (let i = 1; i <= count; i++) {
                if(!store[i]) store[i] = { title: '', link: '', cta: 'LEARN_MORE', img: null, file: null };
                store[i].link = standardUrl;
                const input = document.getElementById(`link-input-${i}`);
                if (input) input.value = standardUrl;
            }
            addLog("Link Protocol được đồng bộ chuẩn ID", "success");
        }

        function initGrid() {
            const count = parseInt(document.getElementById('card-count').value);
            const container = document.getElementById('card-grid');
            container.innerHTML = '';
            const sel = document.getElementById('p-select');
            const currentLink = sel.value ? generateStandardLink(sel.value) : (store[1]?.link || '');

            for(let i=1; i<=count; i++) {
                if (!store[i]) store[i] = { title: '', link: currentLink, cta: 'LEARN_MORE', img: null, file: null };
                const data = store[i];
                const div = document.createElement('div');
                div.className = "glass-panel p-4 flex gap-4 neon-border group items-center";
                div.innerHTML = `
                    <div class="w-20 h-20 shrink-0 bg-white/5 rounded-2xl flex items-center justify-center overflow-hidden cursor-pointer border border-white/10 relative" onclick="document.getElementById('file-${i}').click()">
                        <img id="img-${i}" src="${data.img || ''}" class="${data.img ? '' : 'hidden'} absolute inset-0 w-full h-full object-cover z-10">
                        <span class="text-[8px] font-black text-slate-600 z-0 uppercase">Thẻ ${i}</span>
                    </div>
                    <div class="flex-1 space-y-3">
                        <input type="text" placeholder="Tiêu đề..." oninput="updateStore(${i}, 'title', this.value)" value="${data.title}" class="w-full bg-transparent border-b border-white/10 text-xs font-bold text-white outline-none focus:border-sky-500 py-0.5">
                        <input type="text" id="link-input-${i}" placeholder="Target URL..." oninput="updateStore(${i}, 'link', this.value)" value="${data.link}" class="w-full bg-transparent border-b border-white/10 text-[9px] font-mono text-sky-400 outline-none">
                        <select onchange="updateStore(${i}, 'cta', this.value)" class="w-full bg-white/5 text-[9px] font-black py-1.5 px-3 rounded-lg text-slate-300 outline-none">
                            <option value="LEARN_MORE" ${data.cta === 'LEARN_MORE' ? 'selected' : ''}>XEM CHI TIẾT</option>
                            <option value="MESSAGE_PAGE" ${data.cta === 'MESSAGE_PAGE' ? 'selected' : ''}>GỬI TIN NHẮN</option>
                            <option value="LIKE_PAGE" ${data.cta === 'LIKE_PAGE' ? 'selected' : ''}>THÍCH TRANG</option>
                            <option value="SHOP_NOW" ${data.cta === 'SHOP_NOW' ? 'selected' : ''}>MUA NGAY</option>
                            <option value="BOOK_NOW" ${data.cta === 'BOOK_NOW' ? 'selected' : ''}>ĐẶT NGAY</option>
                            <option value="CALL_ME" ${data.cta === 'CALL_ME' ? 'selected' : ''}>GỌI CHO TÔI</option>
                            <option value="BOOK_TRAVEL" ${data.cta === 'BOOK_TRAVEL' ? 'selected' : ''}>ĐẶT CHUYẾN ĐI</option>
                            <option value="NO_BUTTON" ${data.cta === 'NO_BUTTON' ? 'selected' : ''}>KHÔNG NÚT</option>
                            <option value="CONTACT_US" ${data.cta === 'CONTACT_US' ? 'selected' : ''}>LIÊN HỆ</option>
                            <option value="INSTALL_APP" ${data.cta === 'INSTALL_APP' ? 'selected' : ''}>CÀI ỨNG DỤNG</option>
                            <option value="OPEN_LINK" ${data.cta === 'OPEN_LINK' ? 'selected' : ''}>MỞ LIÊN KẾT</option>
                            <option value="PLAY_GAME" ${data.cta === 'PLAY_GAME' ? 'selected' : ''}>CHƠI GAME</option>
                        </select>
                    </div>
                    <input type="file" id="file-${i}" class="hidden" accept="image/*" onchange="handleFile(${i}, this)">
                `;
                container.appendChild(div);
            }
        }

        async function connectSystem() {
            const tk = document.getElementById('tk-input').value.trim();
            const btn = document.getElementById('btn-conn');
            if(!tk) return alert("Vui lòng nhập Token!");
            btn.innerText = "AUTHENTICATING...";
            try {
                const me = await fetch(`https://graph.facebook.com/me?fields=name,id&access_token=${tk}`).then(r => r.json());
                const pages = await fetch(`https://graph.facebook.com/me/accounts?fields=name,id,access_token&access_token=${tk}`).then(r => r.json());
                if (pages.data) {
                    document.getElementById('user-display').classList.remove('hidden');
                    document.getElementById('user-greeting').innerText = me.name;
                    const sel = document.getElementById('p-select');
                    sel.innerHTML = pages.data.map(p => `<option value="${p.id}" data-token="${p.access_token}">${p.name}</option>`).join('');
                    addLog(`Đã xác thực Agent: ${me.name}`, "success");
                    syncAllLinks();
                }
            } catch(e) { addLog("Lỗi: Token không hợp lệ", "error"); }
            btn.innerText = "Khởi tạo kết nối";
        }

        function handleFile(id, input) {
            const file = input.files[0];
            if(file) {
                const reader = new FileReader();
                reader.onload = (e) => {
                    store[id].img = e.target.result; store[id].file = file;
                    document.getElementById(`img-${id}`).src = e.target.result;
                    document.getElementById(`img-${id}`).classList.remove('hidden');
                };
                reader.readAsDataURL(file);
            }
        }

        function updateStore(id, field, value) {
            if(!store[id]) store[id] = { title: '', link: '', cta: 'LEARN_MORE', img: null, file: null };
            store[id][field] = value;
            if (field === 'link') {
                const count = parseInt(document.getElementById('card-count').value);
                for (let i = 1; i <= count; i++) {
                    if (!store[i]) store[i] = { title: '', link: value, cta: 'LEARN_MORE', img: null, file: null };
                    store[i].link = value;
                    const input = document.getElementById(`link-input-${i}`);
                    if (input) input.value = value;
                }
            }
        }

        async function publish() {
            const sel = document.getElementById('p-select');
            if(!sel.value) return alert("Chọn Fanpage trước!");
            
            const mode = document.getElementById('post-mode').value;
            const schedTime = document.getElementById('schedule-time').value;
            let unixTime = null;

            if(mode === 'SCHEDULE') {
                if(!schedTime) return alert("Vui lòng chọn thời gian lên lịch!");
                unixTime = Math.floor(new Date(schedTime).getTime() / 1000);
                if(unixTime < (Date.now()/1000) + 600) return alert("Thời gian lên lịch phải cách hiện tại ít nhất 10 phút!");
            }

            const pid = sel.value;
            const ptk = sel.options[sel.selectedIndex].dataset.token;
            const btn = document.getElementById('btn-pub');
            const activeIds = Object.keys(store).filter(id => store[id].file);
            if(activeIds.length < 2) return alert("Tối thiểu 2 ảnh!");
            
            btn.innerText = "PROCESSING..."; btn.disabled = true;
            try {
                let cards = [];
                for(let id of activeIds) {
                    const fd = new FormData();
                    fd.append('source', store[id].file); fd.append('published', 'false'); fd.append('access_token', ptk);
                    const photo = await fetch(`https://graph.facebook.com/${pid}/photos`, { method: 'POST', body: fd }).then(r => r.json());
                    const detail = await fetch(`https://graph.facebook.com/${photo.id}?fields=images&access_token=${ptk}`).then(r => r.json());
                    cards.push({ 
                        media_fbid: photo.id, 
                        name: store[id].title, 
                        link: store[id].link, 
                        picture: detail.images[0].source, 
                        call_to_action: { type: store[id].cta, value: { link: store[id].link } } 
                    });
                }

                const postData = { 
                    message: document.getElementById('caption').value, 
                    child_attachments: cards, 
                    link: cards[0].link, 
                    access_token: ptk 
                };

                if(mode === 'SCHEDULE') {
                    postData.published = false;
                    postData.scheduled_publish_time = unixTime;
                }

                const res = await fetch(`https://graph.facebook.com/${pid}/feed`, { 
                    method: 'POST', 
                    headers: { 'Content-Type': 'application/json' }, 
                    body: JSON.stringify(postData) 
                }).then(r => r.json());

                if(res.id) {
                    const logMsg = mode === 'SCHEDULE' ? "Đã lên lịch thành công" : "Đăng bài thành công";
                    addLog(logMsg, mode === 'SCHEDULE' ? 'schedule' : 'success', `https://www.facebook.com/${res.id}`);
                    alert(mode === 'SCHEDULE' ? "Đã lên lịch phát hành!" : "Đăng bài thành công!");
                } else {
                    throw res.error;
                }
            } catch(e) { 
                console.error(e);
                addLog("Lỗi thực thi Engine", "error"); 
                alert("Lỗi: " + (e.message || "Không thể đăng bài"));
            }
            btn.innerText = "Xác nhận phát hành hệ thống"; btn.disabled = false;
        }

        initGrid();
    </script>
